CLASS CacodemonBase : Cacodemon
{
	Action state cacomeleecheck()
	{
		If(target)
		{
			FLineTraceData LOF;
		
			LineTrace(AngleTo(Target),400,0,TRF_SOLIDACTORS,0,data:LOF);
		
			if (LOF.HitActor == Target && Distance3D(target) < 290)
			{
				return resolvestate("melee");
			}
			Else
			{
				Return resolvestate(null);
			}
		}
		
		Else
		{
			Return resolvestate(null);
		}
	}
}
